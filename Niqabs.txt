    private void table_records() {
        int count;
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost/items", "items_user", "items_password");
            PreparedStatement pst = con.prepareStatement("select * from items_details");
            ResultSet rs = pst.executeQuery();
            ResultSetMetaData md = rs.getMetaData();
            count = md.getColumnCount();
            DefaultTableModel dtm = (DefaultTableModel) jTable2.getModel();
            dtm.setRowCount(0);
            while (rs.next()) {
                Vector<Object> v = new Vector<>();
                for (int i = 1; i <= count; i++) {
                    v.add(rs.getObject(i));
                }
                dtm.addRow(v);
            }
        } catch (ClassNotFoundException | SQLException ex) {
            Logger.getLogger(hijabs.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

 private void updateTotalAmount() {
    int rowCount = jTable2.getRowCount();
    double totalAmount = 0.0;
    
    // Sum up the prices in the table
    for (int i = 0; i < rowCount; i++) {
        double price = (int) jTable2.getValueAt(i, 2); // Assuming price is in the 3rd column
        totalAmount += price;
    }
    
    // Update the total amount label
    niqjLabel9.setText("₹ " + totalAmount);
    
    // Insert/update the total amount in the database
    try {
        Class.forName("com.mysql.cj.jdbc.Driver");
        con = DriverManager.getConnection("jdbc:mysql://localhost/items", "items_user", "items_password");
        Statement stmt = con.createStatement();
        ResultSet rs = stmt.executeQuery("SELECT SUM(amount) FROM total_amount");
        rs.next();
        int count = rs.getInt(1);
        if (count == 0) {
            insertTotalAmount((int) totalAmount);
        } else {
            updateTotalAmountInDatabase((int) totalAmount);
        }
    } catch (ClassNotFoundException | SQLException ex) {
        Logger.getLogger(hijabs.class.getName()).log(Level.SEVERE, null, ex);
    }
}

    private void niqdeleteActionPerformed(java.awt.event.ActionEvent evt) {                                          
        DefaultTableModel dtmm = (DefaultTableModel) jTable2.getModel();
        int[] selectedRows = jTable2.getSelectedRows();
UIManager.put("OptionPane.messageFont", new Font("Times New Roman", Font.BOLD, 18));
        UIManager.put("OptionPane.buttonFont", new Font("Times New Roman", Font.BOLD, 16));
        if (selectedRows.length > 0) {
            int confirmation = JOptionPane.showConfirmDialog(this, "Are you sure you want to delete the selected items?", "Delete Items", JOptionPane.YES_NO_OPTION);

            if (confirmation == JOptionPane.YES_OPTION) {
                try {
                    Class.forName("com.mysql.cj.jdbc.Driver");
                     con = DriverManager.getConnection("jdbc:mysql://localhost/items", "items_user", "items_password");

                    for (int i = selectedRows.length - 1; i >= 0; i--) {
                        int row = selectedRows[i];
                        int id = (int) dtmm.getValueAt(row, 0);

                        pst = con.prepareStatement("delete from items_details where id = ?");
                        pst.setInt(1, id);
                        pst.executeUpdate();
                        dtmm.removeRow(row);
                    }
                   
                    reassignIDs();
                    updateTotalAmount();
                } catch (ClassNotFoundException | SQLException ex) {
                    Logger.getLogger(hijabs.class.getName()).log(Level.SEVERE, null, ex);
                }
            }
        } else {
            if(dtmm.getRowCount()==0){
                JOptionPane.showMessageDialog(this, "Your cart is empty!");
            }else{
            JOptionPane.showMessageDialog(this, "No items selected for deletion");}
        }
    }                                         

 private void reassignIDs() {
    try {
        // Reset the auto-increment value to 1
        pst = con.prepareStatement("SET @num := 0");
        pst.executeUpdate();
        
        // Update the IDs with the new sequence
        pst = con.prepareStatement("UPDATE items_details SET id = (@num := @num + 1)");
        pst.executeUpdate();
        
        // Reset the auto-increment value to the next number
        pst = con.prepareStatement("ALTER TABLE items_details AUTO_INCREMENT = 1");
        pst.executeUpdate();
        
    } catch (SQLException ex) {
        Logger.getLogger(hijabs.class.getName()).log(Level.SEVERE, null, ex);
    }
}
